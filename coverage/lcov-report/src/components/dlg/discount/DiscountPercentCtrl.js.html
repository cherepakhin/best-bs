import DiscountPercentDlg from './DiscountPercentDlg';
import { CONST_ACTION, INITVAR } from '../../../constants';
import ACtrlDlg from '../ACtrlDlg';
import fetch from 'isomorphic-fetch';
 
/**
 * Контроллер для управления диалогом ввода процента скидки
 */
class DiscountPercentCtrl extends ACtrlDlg {
  constructor() {
    super();
    this.widget = DiscountPercentDlg;
    this.modalType = CONST_ACTION.DLG_DISCOUNT_PERCENT;
    this.title = 'Введите прооцент скидки';
    this.description = '----';
  }
 
  openDlg(dispatch, modalProps) {
    // Создаю замыкания для функции setTimeout. Там другой this
    // Связывание нужно для работы с dispatch в фунциях ok() и cancel()
    let _ok = this.ok.bind(dispatch);
    let _cancel = this.cancel.bind(dispatch);
    dispatch({
        type: CONST_ACTION.OPEN_DIALOG,
        modalType: this.modalType,
        modalProps: Object.assign(modalProps || {}, {
            title: this.title,
            ok: _ok,
            cancel: _cancel,
            doc: modalProps.doc
          })
      });
  }
 
  ok(doc, nameDiscount, percent) {
    let dispatch = this;
    let param = [{
        name: 'Процент',
        param: percent
      }, ];
    let body = '&doc=' + encodeURIComponent(JSON.stringify(doc)) +
        '&nameDiscount=' + encodeURIComponent(nameDiscount) +
        '&param=' + encodeURIComponent(JSON.stringify(param));
 
    fetch(INITVAR.URL_SERVER + '/discount.ctrl/apply', {
        method: 'POST',
        headers: {
            Accept: 'application/json, text/plain, */*',
            'Content-type': 'application/x-www-form-urlencoded; charset=UTF-8'
          },
        body: body
 
      }).then(function (response) {
        return response.json();
      }).then(function (json) {
        // console.log('json');
        // console.log(json);
        dispatch({
            type: CONST_ACTION.SET_DOC,
            payload: {
                doc: json
              }
          });
      }).catch(function (ex) {
        console.log('Failed create doc /discount.ctrl/apply', ex);
      });
 
    dispatch({ type: CONST_ACTION.CLOSE_DIALOG });
  }
}
 
// export default DiscountPercentCtrl;
export default new DiscountPercentCtrl();
