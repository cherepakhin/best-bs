import React from 'react';
import { Button, Modal } from 'react-bootstrap';
 
/*
 Сообщение об ошибке
 */
let MessageErrorDlg = React.createClass({
    componentDidMount: function () {
        this.btnOk.focus();
      },
 
    // Подстановка реализации должна быть сделана в dlg.js
    fnOk: function (...args) {
        console.log(args);
      },
 
    handleOk: function () {
        this.fnOk();
      },
 
    handleChange: function (e) {
        if (e.key === 'Enter' || e.key === 'Escape') {
          e.preventDefault();
          this.handleOk();
        }
      },
 
    render: function () {
        return <div className='static-modal'>
            <Modal.Dialog>
                <Modal.Header className='bg-info'>
                    <Modal.Title>Ошибка</Modal.Title>
                </Modal.Header>
 
                <Modal.Body>
                    {this.props.message}
                </Modal.Body>
 
                <Modal.Footer>
                    <Button
                        onClick={this.handleOk}
                        onKeyDown={this.handleChange}
                        inputRef={node => this.btnOk = node}
                        >Закрыть</Button>
                </Modal.Footer>
 
            </Modal.Dialog>
        </div>;
      },
 
    propTypes: {
        message: React.PropTypes.string.isRequired,
      },
  });
 
export default MessageErrorDlg;
