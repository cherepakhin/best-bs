import cancel from './cancel.js';
 
export default class ADlg {
  constructor(typeDlg, widget, fnOk) {
    this.typeDlg = typeDlg;
    this.widget = widget;
 
    this.widget.fnOk = cancel;
    if (fnOk != undefined) {
      this.widget.fnOk = fnOk;
    }
 
    this.widget.fnCancel = cancel;
  }
 
  get dispatch() {
    return this._dispatch;
  }
 
  set dispatch(dispatch) {
    this._dispatch = dispatch;
    this.widget.fnOk = this.widget.fnOk.bind(this._dispatch);
    this.widget.fnCancel = this.widget.fnCancel.bind(this._dispatch);
  }
}
