export let del = function (payment) {
  return (dispatch) => {
    let body = 'n=' + encodeURIComponent(payment.n);
    return fetch(INITVAR.URL_SERVER + '/payment.ctrl/delete', {
      method: 'POST',
      headers: {
        Accept: 'application/json, text/plain, */*',
        'Content-type': 'application/x-www-form-urlencoded; charset=UTF-8'
      },
      body: body
    }).then(function (response) {
      return response.json();
    }).then(function (json) {
      dispatch({
        type: CONST_ACTION.SET_DOC,
        payload: json
      });
    }).catch(function (ex) {
      let err = 'Failed create doc /payment.ctrl/delete.' + ex.message;
      console.log(err);
      mapDlg.get(CONST_ACTION.DLG_MESSAGE)
      .show(dispatch, err);
    });
  };
};
