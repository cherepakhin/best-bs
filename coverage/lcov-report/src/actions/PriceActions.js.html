import { CONST_ACTION, INITVAR } from '../constants';
import fetch from 'isomorphic-fetch';
 
export function getByFeature(nameGroupT, queryFeature, onRest) {
  return (dispatch) => {
      dispatch({
          type: CONST_ACTION.PROGRESS_SHOW,
        });
      let body = 'groupt=' + encodeURIComponent(nameGroupT) +
          '&filter=' + encodeURIComponent(JSON.stringify(queryFeature)) +
          '&onrest=' + encodeURIComponent(onRest);
      return fetch(INITVAR.URL_SERVER + '/price.ctrl/getByFilter', {
          method: 'POST',
          headers: {
              Accept: 'application/json, text/plain, */*',
              'Content-type': 'application/x-www-form-urlencoded; charset=UTF-8'
            },
          body: body
 
        }).then(function (response) {
          return response.json();
        }).then(function (json) {
          // console.log(json);
          dispatch({
              type: CONST_ACTION.GET_GOODS,
              payload: {
                  goods: json
                },
            });
          dispatch({
              type: CONST_ACTION.PROGRESS_HIDE
            });
        }).catch(function (ex) {
          console.log('Failed get goods /price.ctrl/list', ex);
        });
 
    };
}
